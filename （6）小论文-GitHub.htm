        // 初始化图表

        function initCharts() {

            const factorsCtx = document.getElementById('factorsChart').getContext('2d');

            const survivalCtx = document.getElementById('survivalChart').getContext('2d');

           

            factorsChart = new Chart(factorsCtx, {

                type: 'bar',

                data: {

                    labels: ['年龄≥60', '男性', '远处转移', '晚期分期'],

                    datasets: [{

                        label: '得分',

                        data: [0, 0, 0, 0],

                        backgroundColor: ['#ff6b6b', '#4ecdc4', '#45b7d1', '#96ceb4']

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        title: {

                            display: true,

                            text: '各因素得分贡献'

                        },

                        legend: {

                            display: false

                        }

                    },

                    scales: {

                        y: {

                            beginAtZero: true,

                            max: 120

                        }

                    }

                }

            });

           

            survivalChart = new Chart(survivalCtx, {

                type: 'line',

                data: {

                    labels: ['1年', '2年', '3年', '4年', '5年'],

                    datasets: [{

                        label: '生存概率',

                        data: [100, 100, 100, 100, 100],

                        borderColor: '#3498db',

                        backgroundColor: 'rgba(52, 152, 219, 0.1)',

                        tension: 0.4,

                        fill: true

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        title: {

                            display: true,

                            text: '预测生存曲线'

                        }

                    },

                    scales: {

                        y: {

                            min: 0,

                            max: 100,

                            title: {

                                display: true,

                                text: '生存概率 (%)'

                            }

                        }

                    }

                }

            });

        }

       

        function calculatePrognosis() {

            // 获取输入值

            const ageScore = parseInt(document.getElementById('ageGroup').value);

            const genderScore = parseInt(document.getElementById('gender').value);

            const metastasisScore = parseInt(document.getElementById('metastasis').value);

            const stageScore = parseInt(document.getElementById('stage').value);

           

            // 计算总分

            const totalScore = ageScore + genderScore + metastasisScore + stageScore;

           

            // 计算生存概率（基于逻辑函数）

            const survivalProbability = 100 / (1 + Math.exp(-(1.2 - totalScore/150)));

           

            // 风险分层

            let riskLevel, riskColor, recommendation;

            if (totalScore <= 50) {

                riskLevel = "低风险";

                riskColor = "#28a745";

                recommendation = `

                    <h3>💡 临床建议 - 低风险</h3>

                    <ul>

                        <li><strong>建议标准治疗方案</strong></li>

                        <li><strong>常规随访监测</strong>（6-12个月）</li>

                        <li><strong>良好的预后预期</strong></li>

                        <li><strong>5年生存率约65%</strong></li>

                    </ul>

                `;

            } else if (totalScore <= 120) {

                riskLevel = "中风险";

                riskColor = "#ffc107";

                recommendation = `

                    <h3>💡 临床建议 - 中风险</h3>

                    <ul>

                        <li><strong>建议加强辅助治疗</strong></li>

                        <li><strong>密切随访</strong>（3-6个月）</li>

                        <li><strong>考虑个体化治疗调整</strong></li>

                        <li><strong>5年生存率约45%</strong></li>

                    </ul>

                `;

            } else {

                riskLevel = "高风险";

                riskColor = "#dc3545";

                recommendation = `

                    <h3>💡 临床建议 - 高风险</h3>

                    <ul>

                        <li><strong>建议强化治疗方案</strong></li>

                        <li><strong>密切随访</strong>（1-3个月）</li>

                        <li><strong>考虑临床试验参与</strong></li>

                        <li><strong>积极支持治疗</strong></li>

                        <li><strong>5年生存率约20%</strong></li>

                    </ul>

                `;

            }

           

            // 更新显示

            document.getElementById('scoreDisplay').textContent = totalScore;

            document.getElementById('survivalDisplay').textContent = survivalProbability.toFixed(1) + '%';

            document.getElementById('riskLevelDisplay').textContent = riskLevel;

            document.getElementById('riskLevelDisplay').style.color = riskColor;

           

            // 更新风险指示器位置 (0-270分对应0-100%)

            const markerPosition = (totalScore / 270) * 100;

            document.getElementById('riskMarker').style.left = markerPosition + '%';

           

            // 更新图表

            factorsChart.data.datasets[0].data = [ageScore, genderScore, metastasisScore, stageScore];

            factorsChart.update();

           

            // 根据风险等级更新生存曲线

            let survivalData;

            if (riskLevel === "低风险") {

                survivalData = [95, 90, 85, 80, 75];

            } else if (riskLevel === "中风险") {

                survivalData = [85, 75, 65, 55, 45];

            } else {

                survivalData = [70, 55, 40, 30, 20];

            }

           

            survivalChart.data.datasets[0].data = survivalData;

            survivalChart.update();

           

            // 更新推荐内容

            document.getElementById('recommendation').innerHTML = recommendation;

            document.getElementById('recommendation').className = 'recommendation ' +

                (riskLevel === "低风险" ? "low-risk" :

                 riskLevel === "中风险" ? "medium-risk" : "high-risk");

           

            // 显示结果面板

            document.getElementById('results').style.display = 'block';

            document.getElementById('placeholder').style.display = 'none';

        }

       

        // 页面加载时初始化

        document.addEventListener('DOMContentLoaded', function() {

            initCharts();

        });
